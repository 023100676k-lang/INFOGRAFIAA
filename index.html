    @Nombre VARCHAR(50),
    @Apellido VARCHAR(50),
    @DNI CHAR(8),
    @Direccion VARCHAR(100),
    @Email VARCHAR(80)
AS
BEGIN
    IF NOT EXISTS (SELECT 1 FROM Alumno WHERE DNI = @DNI)
        INSERT INTO Alumno (Nombre, Apellido, DNI, Direccion, Email)
        VALUES (@Nombre, @Apellido, @DNI, @Direccion, @Email);
    ELSE
        RAISERROR('El alumno con ese DNI ya existe', 16, 1);
END;
